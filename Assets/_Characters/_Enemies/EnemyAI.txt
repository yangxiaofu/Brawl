tree("Root")
	parallel
		repeat mute tree("BeAlive")
		repeat mute tree("Dead")

tree("Dead")
	while EnemyIsDead
		Succeed

tree("BeAlive")
	while not EnemyIsDead
		fallback
			tree("AttackPlayer")
			tree("SearchForPlayer")
			tree("BeIdle")

tree("SearchForPlayer")
	sequence 
		FindPlayer
		PlayerIsWithinMoveRadius	
		tree("WalkToPlayer")
		not PlayerIsWithinStoppingDistance


tree("WalkToPlayer")
	sequence
		SetWalkAnimation
		MoveToPlayer
		PlayerIsWithinMoveRadius

tree("BeIdle")
	sequence
		SetIdleAnimation
		StayPut
