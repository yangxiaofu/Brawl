tree("Root")
	while IsBot
		parallel
			repeat mute tree("BeAlive")
			repeat mute tree("Kill")

tree("Kill")
	while IsDead
		Succeed

tree("BeAlive")
	while not IsDead
		parallel
			tree("RandomMovementSelection")
			fallback
				tree("Attack")
				tree("ScanForCharacters")
				tree("ScanForSpawners")
				tree("RunForCover")
				tree("Celebrate")

tree("ScanForSpawners")
	sequence
		LowOnAmmo
		ScanForNearbySpawner
		GoToSpawner

tree("Attack")
	sequence
		not IsInDanger
		not LowOnAmmo
		DoCharactersExist
		ChooseRandomTarget
		IsTargetWithinShootDistance
		tree("ShootTarget")

tree("ScanForCharacters")
	sequence
		not IsInDanger
		not LowOnAmmo
		DoCharactersExist
		ChooseRandomTarget
		tree("MoveTowardTarget")

tree("MoveTowardTarget")
	repeat 5.0
		sequence
			MoveTowardTarget	
			IsTargetWithinShootDistance
			tree("ShootTarget")

tree("ShootTarget")
	sequence
		repeat 3.0
			sequence
				not IsInDanger
				IsTargetWithinShootDistance
				ShootTarget
	
tree("RunForCover")
	sequence
		MoveToRandomSpot
		repeat
			tree("RecoverHealth")
			
tree("RecoverHealth")
	sequence
		not IsBeingAttack
		not ReadyToFight

tree("Celebrate")
	Succeed
